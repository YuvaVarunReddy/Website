    Example:

    <!DOCTYPE html>

    <html>

    <head>

    <title>CSS Layout - Horizontal & Vertical Align</title>

    <style>

    .center { margin: auto; width: 60%;
    border: 3px solid #73AD21;
    padding: 10px; }

    .center1 { text-align: center;
    border: 3px solid green;
    }

    .img1 { display: block;
    margin-left: auto;
    margin-right: auto;
    justify-content: center;
    align-items: center;
    }

    .right { justify-content: right; width: 300px;
    float: right;
    border: 3px solid #73AD21;
    padding: 10px;
    }

    .right1 { float: right;
    width: 300px;
    border: 3px solid #73AD21;
    padding: 10px;
    }

    .in1 { display: inline-block; }

    .in { border: 3px solid #4CAF50;
    padding: 5px; }

    .img3 { float: right; }

    .img4 { float: right; }

    .clearfix::after { content: "";
    clear: both;
    display: table;
    }

    .center2 { padding: 70px 0;
    border: 3px solid green;
    }

    .center3 { padding: 70px 0;
    border: 3px solid green;
    text-align: center;
    }

    .center4 { line-height: 200px;
    height: 200px;
    border: 3px solid green;
    text-align: center;
    }

    .center5 { line-height: 1.5;
    display: inline-block;
    vertical-align: middle;
    }

    .center6 { height: 200px;
    position: relative;
    border: 3px solid green;
    }

    .center7 { margin: 0;
    position: absolute;
    top: 50%; left: 50%;
    -ms-transform: translate(-50%, -50%);
    transform: translate(-50%, -50%);
    }

    .center8 { display: flex;
    justify-content: center;
    align-items: center;
    height: 200px;
    border: 3px solid green;
    }

    </style>

    </head>
