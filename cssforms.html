    CSS Forms

    Example:

    <!DOCTYPE html>

    <html>

    <head>

    <title>Forms</title>

    <style>

    .input111 { width: 100%; }

    .in[type=text] { width: 100%;
    padding: 12px 20px; margin: 8px 0;
    box-sizing: border-box; }

    .in1[type=text] { width: 100%;
    padding: 12px 20px; margin: 8px 0;
    box-sizing: border-box; border: 2px solid red;
    border-radius: 4px; }

    .in2[type=text] { width: 100%;
    padding: 12px 20px; margin: 8px 0;
    box-sizing: border-box; border: none;
    border-bottom: 2px solid red; }

    .in3[type=text] { width: 100%;
    padding: 12px 20px; margin: 8px 0;
    box-sizing: border-box; border: none;
    background-color: #3CBC8D; color: white;
    }

    .in4[type=text] { width: 130px;
    box-sizing: border-box; border: 2px solid #ccc;
    border-radius: 4px; font-size: 16px;
    background-color: white;
    background-image: url('searchicon.png');
    background-position: 10px 10px;
    background-repeat: no-repeat;
    padding: 12px 20px 12px 40px;
    transition: width 0.4s ease-in-out; }

    .in5[type=text]:focus { width: 100%; }

    textarea { width: 100%;
    height: 150px; padding: 12px 20px;
    box-sizing: border-box; border: 2px solid #ccc;
    border-radius: 4px;
    background-color: #f8f8f8;
    font-size: 16px;
    resize: none; }

    select { width: 100%;
    padding: 16px 20px; border: none;
    border-radius: 4px;
    background-color: #f1f1f1; }

    input[type=button], input[type=submit],
    input[type=reset] { background-color: #04AA6D;
    border: none; color: white;
    padding: 16px 32px; text-decoration: none;
    margin: 4px 2px; cursor: pointer;
    }

    </style>

    </head>

    <body>

    <h2>A full-width input field</h2>

    <form>
    <label for="fname">First Name</label>
    <input type="text" id="fname" name="fname" class="input111">
    </form>

    <h2>Padded input fields</h2>

    <form>
    <label for="fname">First Name</label>
    <input type="text" id="fname" name="fname" class="in">
    <label for="lname">Last Name</label>
    <input type="text" id="lname" name="lname" class="in">
    </form>

    <h2>Input fields with borders</h2>

    <form>
    <label for="fname">First Name</label>
    <input type="text" id="fname" name="fname" class="in1">
    <label for="lname">Last Name</label>
    <input type="text" id="lname" name="lname" class="in1">
    </form>

    <h2>Input fields with bottom border</h2>

    <form>
    <label for="fname">First Name</label>
    <input type="text" id="fname" name="fname" class="in2">
    <label for="lname">Last Name</label>
    <input type="text" id="lname" name="lname" class="in2">
    </form>

    <h2>Input fields with color</h2>

    <form>
    <label for="fname">First Name</label>
    <input type="text" id="fname" name="fname" value="John" class="in3">
    <label for="lname">Last Name</label>
    <input type="text" id="lname" name="lname" value="Doe" class="in3">
    </form>

    <h2>Animate width of search input</h2>

    <form>
    <input type="text" name="search" placeholder="Search.." class="in4 in5">
    </form>

    <h2>Styling textarea</h2>

    <p><strong>Tip:</strong> Use the resize
    to prevent textareas from being resized (disable the "grabber" in the bottom right corner):</p>

    <form>
    <textarea>Some text...</textarea>
    </form>

    <h2>Styling a select menu</h2>

    <form>
    <select id="country" name="country">
    <option value="au">Australia</option>
    <option value="ca">Canada</option>
    <option value="usa">USA</option>
    </select>
    </form>

    <h2>Styling form buttons</h2>

    <input type="button" value="Button">
    <input type="reset" value="Reset">
    <input type="submit" value="Submit">

    </body>

    </html>
